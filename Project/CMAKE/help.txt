build in windows:
cmake -DCMAKE_TOOLCHAIN_FILE=STM32F0.cmake -DCMAKE_BUILD_TYPE=DEBUG -G "MinGW Makefiles" CMakeLists.txt -B STM32F0
cmake -DCMAKE_BUILD_TYPE=DEBUG CMakeLists.txt -G "MinGW Makefiles" -B WIN32

build in linux:
cmake -DCMAKE_TOOLCHAIN_FILE=STM32F0.cmake -DCMAKE_BUILD_TYPE=DEBUG CMakeLists.txt -B STM32F0
cmake -DCMAKE_BUILD_TYPE=DEBUG CMakeLists.txt -B LINUX
